packages = ["plotly", "numpy"] from pyscript import document, display import numpy as np import plotly.graph_objects as go from collections import Counter import math # ============================================================ # UTILIDADES # ============================================================ def parsear_datos(texto): texto = texto.replace(",", " ") return [float(x) for x in texto.split()] # ============================================================ # TABLA DE FRECUENCIAS SIMPLE # ============================================================ def tabla_html(valores, frec): total = sum(frec) acum = 0 filas = "" for v, f in zip(valores, frec): acum += f hi = f / total Hi = acum / total filas += f""" {v:g} {f} {hi:.3f} {acum} {Hi:.3f} """ return f""" {filas}
Valor Frecuencia (fi) Frecuencia Relativa (hi) Frecuencia Acumulada (Fi) Frecuencia Relativa Acumulada (Hi)
Total {total} 1.000 {total} 1.000
""" # ============================================================ # TABLA DE FRECUENCIAS AGRUPADA (CORREGIDA) # ============================================================ def tabla_agrupada_html(datos): n = len(datos) k = round(1 + 3.322 * math.log10(n)) # Regla de Sturges frecuencias, bordes = np.histogram(datos, bins=k) filas = "" total = frecuencias.sum() acum = 0 for i in range(len(frecuencias)): li = bordes[i] ls = bordes[i + 1] fi = int(frecuencias[i]) acum += fi hi = fi / total Hi = acum / total intervalo = f"[{li:.2f}, {ls:.2f})" if i < len(frecuencias)-1 else f"[{li:.2f}, {ls:.2f}]" filas += f""" {i+1} {intervalo} {fi} {hi:.3f} {acum} {Hi:.3f} """ return f""" {filas}
Clase Intervalo Frecuencia hi Fi Hi
Total {total} 1.000 {total} 1.000
""" # ============================================================ # RESUMEN ESTADÍSTICO # ============================================================ def resumen_html(datos): moda = Counter(datos).most_common(1)[0][0] q1 = np.percentile(datos, 25) q3 = np.percentile(datos, 75) return f"""
n: {len(datos)}
Mínimo: {min(datos):.2f}
Máximo: {max(datos):.2f}
Media: {np.mean(datos):.2f}
Mediana: {np.median(datos):.2f}
Moda: {moda:.2f}
Desv. Estándar: {np.std(datos, ddof=1):.2f}
Varianza: {np.var(datos, ddof=1):.2f}
Q1: {q1:.2f}
Q3: {q3:.2f}
""" # ============================================================ # FUNCIÓN PRINCIPAL # ============================================================ def calcular(event=None): texto = document.getElementById("datos").value.strip() if not texto: return datos = parsear_datos(texto) datos.sort() # Frecuencia simple conteo = Counter(datos) valores = sorted(conteo.keys()) frecuencias = [conteo[v] for v in valores] document.getElementById("tabla-frecuencias").innerHTML = tabla_html(valores, frecuencias) # Frecuencia agrupada (si existe el contenedor) try: document.getElementById("tabla-frecuencias-agrupada").innerHTML = tabla_agrupada_html(datos) except: pass # Resumen document.getElementById("resumen").innerHTML = resumen_html(datos) # Histograma fig1 = go.Figure(go.Histogram(x=datos, nbinsx=7, marker_color="#2f6fb2")) fig1.update_layout(template="plotly_white", height=320) display(fig1, target="histograma", append=False) # Boxplot fig2 = go.Figure(go.Box(x=datos, orientation='h', marker_color="#2f6fb2")) fig2.update_layout(template="plotly_white", height=320) display(fig2, target="boxplot", append=False) # Datos iniciales document.getElementById("datos").value = "12, 15, 18, 20, 22, 12, 14, 16, 18, 21, 23, 25, 17, 19, 22, 24, 26, 28, 30, 16, 14, 18" calcular()
